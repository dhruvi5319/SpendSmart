# Database Migration

Create and manage PostgreSQL database migrations using Alembic.

## Migration Workflow

### 1. Plan Schema Changes

Before creating a migration, document:
- What tables are being added/modified?
- What columns are changing?
- Are there any data migrations needed?
- What indexes should be added?
- Any foreign key relationships?

### 2. Update SQLAlchemy Model

First, update the model in `apps/api/app/db/models/`

```python
# apps/api/app/db/models/{model}.py
from sqlalchemy import Column, String, ForeignKey, DateTime, Numeric, Boolean, Text, Index
from sqlalchemy.dialects.postgresql import UUID, JSONB
from sqlalchemy.orm import relationship
from datetime import datetime
import uuid

from app.db.base import Base

class {Model}(Base):
    __tablename__ = "{table_name}"

    id = Column(UUID(as_uuid=True), primary_key=True, default=uuid.uuid4)
    user_id = Column(UUID(as_uuid=True), ForeignKey("users.id"), nullable=False)

    # Your columns
    name = Column(String(100), nullable=False)
    amount = Column(Numeric(12, 2), nullable=False)
    currency = Column(String(3), default="USD")
    description = Column(Text)
    is_active = Column(Boolean, default=True)
    metadata = Column(JSONB, default={})

    # Timestamps
    created_at = Column(DateTime, default=datetime.utcnow)
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)

    # Relationships
    user = relationship("User", back_populates="{table_name}")

    # Indexes
    __table_args__ = (
        Index('ix_{table_name}_user_id', 'user_id'),
        Index('ix_{table_name}_created_at', 'created_at'),
        Index('ix_{table_name}_user_date', 'user_id', 'created_at'),
    )
```

### 3. Generate Migration

```bash
cd apps/api

# Auto-generate migration from model changes
alembic revision --autogenerate -m "descriptive_migration_name"
```

### 4. Review Generated Migration

Check `apps/api/alembic/versions/{revision}_descriptive_migration_name.py`

```python
"""descriptive_migration_name

Revision ID: abc123
Revises: xyz789
Create Date: 2024-01-01 12:00:00.000000

"""
from alembic import op
import sqlalchemy as sa
from sqlalchemy.dialects import postgresql

# revision identifiers
revision = 'abc123'
down_revision = 'xyz789'
branch_labels = None
depends_on = None

def upgrade() -> None:
    # ### commands auto generated by Alembic ###
    op.create_table(
        '{table_name}',
        sa.Column('id', postgresql.UUID(as_uuid=True), nullable=False),
        sa.Column('user_id', postgresql.UUID(as_uuid=True), nullable=False),
        sa.Column('name', sa.String(length=100), nullable=False),
        sa.Column('amount', sa.Numeric(precision=12, scale=2), nullable=False),
        sa.Column('currency', sa.String(length=3), nullable=True),
        sa.Column('description', sa.Text(), nullable=True),
        sa.Column('is_active', sa.Boolean(), nullable=True),
        sa.Column('metadata', postgresql.JSONB(astext_type=sa.Text()), nullable=True),
        sa.Column('created_at', sa.DateTime(), nullable=True),
        sa.Column('updated_at', sa.DateTime(), nullable=True),
        sa.ForeignKeyConstraint(['user_id'], ['users.id'], ),
        sa.PrimaryKeyConstraint('id')
    )
    op.create_index('ix_{table_name}_user_id', '{table_name}', ['user_id'])
    op.create_index('ix_{table_name}_created_at', '{table_name}', ['created_at'])
    # ### end Alembic commands ###

def downgrade() -> None:
    # ### commands auto generated by Alembic ###
    op.drop_index('ix_{table_name}_created_at', table_name='{table_name}')
    op.drop_index('ix_{table_name}_user_id', table_name='{table_name}')
    op.drop_table('{table_name}')
    # ### end Alembic commands ###
```

### 5. Apply Migration

```bash
# Apply to database
alembic upgrade head

# Or apply specific revision
alembic upgrade abc123

# Check current revision
alembic current

# View migration history
alembic history
```

## Common Migration Patterns

### Add Column

```python
def upgrade() -> None:
    op.add_column('expenses', sa.Column('receipt_url', sa.String(500), nullable=True))

def downgrade() -> None:
    op.drop_column('expenses', 'receipt_url')
```

### Add Column with Default (for existing rows)

```python
def upgrade() -> None:
    # Add column as nullable first
    op.add_column('expenses', sa.Column('is_household', sa.Boolean(), nullable=True))

    # Set default for existing rows
    op.execute("UPDATE expenses SET is_household = false WHERE is_household IS NULL")

    # Now make it non-nullable
    op.alter_column('expenses', 'is_household', nullable=False, server_default='false')

def downgrade() -> None:
    op.drop_column('expenses', 'is_household')
```

### Add Index

```python
def upgrade() -> None:
    op.create_index('ix_expenses_category_id', 'expenses', ['category_id'])
    # Composite index
    op.create_index('ix_expenses_user_date', 'expenses', ['user_id', 'expense_date'])

def downgrade() -> None:
    op.drop_index('ix_expenses_user_date', table_name='expenses')
    op.drop_index('ix_expenses_category_id', table_name='expenses')
```

### Add Foreign Key

```python
def upgrade() -> None:
    op.add_column('expenses', sa.Column('card_id', postgresql.UUID(as_uuid=True), nullable=True))
    op.create_foreign_key(
        'fk_expenses_card_id',
        'expenses', 'credit_cards',
        ['card_id'], ['id'],
        ondelete='SET NULL'
    )

def downgrade() -> None:
    op.drop_constraint('fk_expenses_card_id', 'expenses', type_='foreignkey')
    op.drop_column('expenses', 'card_id')
```

### Rename Column

```python
def upgrade() -> None:
    op.alter_column('expenses', 'desc', new_column_name='description')

def downgrade() -> None:
    op.alter_column('expenses', 'description', new_column_name='desc')
```

### Change Column Type

```python
def upgrade() -> None:
    # From String to Text
    op.alter_column('expenses', 'notes',
        existing_type=sa.String(255),
        type_=sa.Text(),
        existing_nullable=True
    )

def downgrade() -> None:
    op.alter_column('expenses', 'notes',
        existing_type=sa.Text(),
        type_=sa.String(255),
        existing_nullable=True
    )
```

### Data Migration

```python
from sqlalchemy.sql import table, column
from sqlalchemy import String

def upgrade() -> None:
    # Create reference to table
    expenses = table('expenses',
        column('category', String),
        column('category_id', postgresql.UUID)
    )

    # Migrate data
    op.execute(
        expenses.update()
        .where(expenses.c.category == 'food')
        .values(category_id='uuid-for-food-category')
    )

def downgrade() -> None:
    pass  # Data migrations often can't be reversed
```

### Create Enum Type

```python
def upgrade() -> None:
    # Create enum type
    expense_source = postgresql.ENUM('manual', 'receipt_scan', 'csv_import', 'splitwise',
                                       name='expense_source')
    expense_source.create(op.get_bind())

    # Add column using enum
    op.add_column('expenses',
        sa.Column('source', sa.Enum('manual', 'receipt_scan', 'csv_import', 'splitwise',
                                     name='expense_source'),
                  nullable=True, server_default='manual')
    )

def downgrade() -> None:
    op.drop_column('expenses', 'source')

    # Drop enum type
    expense_source = postgresql.ENUM('manual', 'receipt_scan', 'csv_import', 'splitwise',
                                       name='expense_source')
    expense_source.drop(op.get_bind())
```

## Row-Level Security (RLS) for Supabase

After creating tables, add RLS policies:

```sql
-- Enable RLS
ALTER TABLE {table_name} ENABLE ROW LEVEL SECURITY;

-- Policy: Users can only see their own data
CREATE POLICY "Users can view own {table_name}"
ON {table_name} FOR SELECT
USING (auth.uid() = user_id);

-- Policy: Users can insert their own data
CREATE POLICY "Users can insert own {table_name}"
ON {table_name} FOR INSERT
WITH CHECK (auth.uid() = user_id);

-- Policy: Users can update their own data
CREATE POLICY "Users can update own {table_name}"
ON {table_name} FOR UPDATE
USING (auth.uid() = user_id);

-- Policy: Users can delete their own data
CREATE POLICY "Users can delete own {table_name}"
ON {table_name} FOR DELETE
USING (auth.uid() = user_id);
```

## Troubleshooting

### Migration Conflicts
```bash
# If you have conflicting heads
alembic heads

# Merge heads
alembic merge -m "merge_heads" head1 head2
```

### Rollback Migration
```bash
# Rollback one step
alembic downgrade -1

# Rollback to specific revision
alembic downgrade xyz789

# Rollback all
alembic downgrade base
```

### View SQL Without Executing
```bash
# Show SQL for upgrade
alembic upgrade head --sql

# Show SQL for specific migration
alembic upgrade abc123 --sql
```
